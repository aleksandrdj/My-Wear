        function initClouds() {

            canvas = document.getElementById('weather-canvas');

            ctx = canvas.getContext('2d');

            resize();

            window.addEventListener('resize', resize);

            for(let i=0; i<5; i++) clouds.push({ x: Math.random()*canvas.width, y: Math.random()*canvas.height*0.5, r: 150+Math.random()*200, v: 0.1+Math.random()*0.2, o: 0.1+Math.random()*0.15 });

            animate();

        }

        function resize() { canvas.width = window.innerWidth; canvas.height = window.innerHeight; }

        function animate() {

            ctx.clearRect(0,0,canvas.width,canvas.height);

            clouds.forEach(c => {

                c.x += c.v; if(c.x-c.r > canvas.width) c.x = -c.r;

                const g = ctx.createRadialGradient(c.x, c.y, 0, c.x, c.y, c.r);

                g.addColorStop(0, `rgba(255,255,255,${c.o})`); g.addColorStop(1, 'rgba(255,255,255,0)');

                ctx.fillStyle = g; ctx.beginPath(); ctx.arc(c.x, c.y, c.r, 0, Math.PI*2); ctx.fill();

            });

            requestAnimationFrame(animate);

        }

        function getIcon(c, d) {

            if(c===0) return { icon: d?'fa-sun':'fa-moon' };

            if(c<=2) return { icon: d?'fa-cloud-sun':'fa-cloud-moon' };

            if(c===3) return { icon: 'fa-cloud' };

            if(c>=61 && c<=67) return { icon: 'fa-cloud-rain' };

            if(c>=71 && c<=77) return { icon: 'fa-snowflake' };

            return { icon: 'fa-cloud' };

        }


        function changeLoc(key) {

            document.getElementById('btn-moscow').classList.toggle('active', key==='moscow');

            document.getElementById('btn-adler').classList.toggle('active', key==='adler');

            document.getElementById('btn-gps').classList.remove('active');

            document.getElementById('city-label').innerText = CITIES[key].name;

            fetchWeather(CITIES[key].lat, CITIES[key].lon);

        }


        function getGPS() {

            if(!navigator.geolocation) return alert("GPS не поддерживается");

            navigator.geolocation.getCurrentPosition(p => {

                document.getElementById('city-label').innerText = "Моя локация";

                fetchWeather(p.coords.latitude, p.coords.longitude);

            });

        }
